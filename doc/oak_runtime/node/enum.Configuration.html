pub enum Configuration {
    LogNode,
    GrpcServerNode {
        address: SocketAddr,
        tls_identity: Identity,
    },
    GrpcClientNode {
        uri: Uri,
        root_tls_certificate: Certificate,
        address: String,
    },
    WasmNode {
        module: Arc<Module>,
    },
    External,
}

A Configuration corresponds to a NodeConfiguration protobuf message, with parsed and validated configuration information.
