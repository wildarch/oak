    #[prost(string, tag="1")]
    pub include: std::string::String,
    /// If non-empty, holds a regular expression of test names to exclude from the
    /// test run.  Tests are only run if they match include and do not match
    /// exclude.
    #[prost(string, tag="2")]
    pub exclude: std::string::String,
}

    }
